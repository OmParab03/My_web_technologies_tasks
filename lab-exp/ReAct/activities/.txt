// Example 1: Counter using useState

import React, { useState } from "react";

function Counter() {
    const [count, setCount] = useState(0);

    return (
        <>
            <h2>Count: {count}</h2>
            <button onClick={() => setCount(count + 1)}>
                Increase
            </button>
        </>
    );
}

export default Counter;


// Example 2: Show/Hide Text

import React, { useState } from "react";

function ToggleText() {
    const [show, setShow] = useState(true);

    return (
        <>
            <button onClick={() => setShow(!show)}>
                Toggle Text
            </button>

            {show && <h2>Hello React State!</h2>}
        </>
    );
}

export default ToggleText;







// Hooks are special functions in React
// that allow functional components
// to use state and lifecycle features.

// Common Hooks:
// 1. useState()
// 2. useEffect()
// 3. useContext()
// 4. useRef()
// 5. useMemo()
// 6. useCallback()


// Example of useEffect Hook

import React, { useState, useEffect } from "react";

function ExampleHook() {
    const [count, setCount] = useState(0);

    useEffect(() => {
        console.log("Component Rendered");
    }, []);

    return (
        <>
            <h2>{count}</h2>
            <button onClick={() => setCount(count + 1)}>
                Click
            </button>
        </>
    );
}

export default ExampleHook;
